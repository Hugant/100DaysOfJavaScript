The moving block

Сегодня мы создадим подвижный блок.


Для начала, нам нужно создать 4 кнопки для управления нашим блоком.

				
html:

				<input type = "button" value = "Up"    onclick = "Up()">   <!-- вверх -->
<input type = "button" value = "Down"  onclick = "Down()"> <!-- вниз -->
<input type = "button" value = "Left"  onclick = "Left()"> <!-- влево -->
<input type = "button" value = "Right" onclick = "Right()"><!-- вправо -->

И сам блок.

				
html:

				<div id = "block"></div>

Только не забудте расположить его в центре экрана.

				
CSS:

				#block
{
    position: absolute;
    top: 50%;
    left: 50%;
    margin-left: -50px;
    margin-top: -50px;
    width: 100px;
    height: 100px;
    background: blue;
    transition-duration: 1s;/* Для плавных переходов */
}

Перейдем же к скрипту.


Мы не будем мучиться с получением стилей элемента block, но если вам интересно, то можете заглянуть сюда.

Метод document.getElementById("id").style.marginTop мы можем изменить занчение свойства margin-top у нашего блока.

Аналогичная ситуация с margin-left.

Не забываем, что присвоить мы должны строку, а не число.

				
JavaScript:

				var marginTop = -100;// Отступ сверху
var marginLeft = -100;// Отступ слева
function Up()
{
    document.getElementById("block").style.marginTop = marginTop - 20 + "px";// Неявное преобразование из числа в строку
    marginTop = marginTop - 20;
}

function Down()
{
    document.getElementById("block").style.marginTop = marginTop + 20 + "px"; 
    marginTop = marginTop + 20;
}

function Left()
{
    document.getElementById("block").style.marginLeft = marginLeft - 20 + "px"; 
    marginLeft = marginLeft - 20; 
}

function Right()
{
    document.getElementById("block").style.marginLeft = marginLeft + 20 + "px"; 
    marginLeft = marginLeft + 20;
}

Но наш блок не имеет ограничений и может улететь очень далеко.

Введем ограничение.

			
JavaScript:

			var marginTop = -100;
var marginLeft = -100;
function Up()
{
    if(marginTop > -360)
    {
        document.getElementById("block").style.marginTop = marginTop - 20 + "px"; 
        marginTop = marginTop - 20;
    }
}

function Down()
{
    if(marginTop < 260)
    {
        document.getElementById("block").style.marginTop = marginTop + 20 + "px"; 
        marginTop = marginTop + 20;
    }
}

function Left()
{
    if(marginLeft > -360)
    {
       document.getElementById("block").style.marginLeft = marginLeft - 20 + "px"; 
       marginLeft = marginLeft - 20; 
    }
}

function Right()
{
    if(marginLeft < 260)
    {
        document.getElementById("block").style.marginLeft = marginLeft + 20 + "px"; 
        marginLeft = marginLeft + 20;
    }
}
