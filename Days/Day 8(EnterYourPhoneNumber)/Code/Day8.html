Начнем, как обычно с html.

Нам нужен только одно поле вывода, в данном случае span.

				
html:

				<span id = "number"></span>

Формат вывода будет выглядить как телефонный номер 8 (823) 430-34-34.

Наш скрипт выполняет две задачи, обрабатывает нажатие клавиши мыши и форматирует число для вывода.

				
JavaScript:

				var number = 0;// счетчик

function formatNumber(num) {
    num = num.toString(10);// Перевод из числа в строку

    for(var i = num.length; i < 11; i++) {
    	num = "0" + num;// заполняем число недостающими нулями, к примеру было 12 стало 00000000012
    }

    var output = num[0] + " " + "(" + num[1] + num[2] + num[3] + ") " + // код зоны
               + num[4] + num[5] + num[6] + "-" + // сам номер
               + num[7] + num[8] + "-" +
               + num[9] + num[10];

    return output;// возвращаем телефонный номер
}

num[0] - здесь происходит обращение к первому елементу массива, так как отчет ведется с нуля.

Например, у нас есть номер 89238933434

				var str = "89238933434";
console.log(str[0]);// 8
console.log(str[3]);// 3
console.log(str[10];// 4
console.log(str[11];// error

Теперь нужно добавить слушатель на весь документ.

				
JavaScript:

				window.onload = function() {// при загрузке документа
    document.getElementById("number").innerHTML = formatNumber(number);// вывести отформатированное число( 0 (000) 000-00-00)

    window.onmousedown = function() {// при нажатии клавиши мыши
    	number++;// увеличить счетчик
    	document.getElementById("number").innerHTML = formatNumber(number);// обновить вывод
    };
};
